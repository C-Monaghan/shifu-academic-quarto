% TITLE PAGE AND FRONT MATTER
% ===========================
% This section creates the title page, abstract, keywords, and (some) front matter

\begin{titlepage}

% Generate the main title block (title, authors, affiliations)
% \let\newpage\relax prevents page break before title in some document classes
{\let\newpage\relax\maketitle}

% Apply title page styling (no page numbers)
\thispagestyle{title}

% Vertical spacing after title block
\vspace{0.15in}

% Optional word count display (from Quarto YAML)
$if(word-count)$
{\centering Word count: $word-count$}
$endif$

% Insert a run in abstract
$if(abstract)$
\begin{abstract}
$abstract$
\end{abstract}
$endif$

% Insert keywords
$if(keywords)$
\begin{keywords}
\def\sep{;\ }
$for(keywords/allbutlast)$$keywords$\sep $endfor$
$for(keywords/last)$$keywords$$endfor$
\end{keywords}
$endif$

\vfill  % Fill the rest of the page with whitespace

% Add article correspondance
$if(correspondence-prefix)$
\flushleft \textbf{Correspondence}

$for(by-author)$
$if(it.attributes.corresponding)$
\par\raggedright\setlength{\parindent}{0in}%
    $correspondence-prefix$~$it.name.literal$$if(it.email)$%
    (\href{mailto:$it.email$}{$it.email$})$endif$$if(it.affiliations)$, %
    $for(it.affiliations)$$if(it.department)$$it.department$, $endif$$it.name$,%
    $if(it.address)$$it.address$, $endif$$if(it.city)$$it.city$, $endif$$if(it.region)$%
    $it.region$ $endif$$if(it.postal-code)$$it.postal-code$$endif$$if(it.country)$, %
    $it.country$$endif$$sep$; $endfor$ $endif$
$endif$
$endfor$
$endif$

\vfill  % Fill the rest of the page with whitespace

\end{titlepage}
